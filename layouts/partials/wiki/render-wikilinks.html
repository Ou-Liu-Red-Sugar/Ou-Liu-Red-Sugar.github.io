{{- /* layouts/partials/wiki/render-wikilinks.html Input : a string of HTML (typically .Content). Output : same string with every wikilink replaced by either - display (resolved) - display (unresolved) Supported wikilink syntaxes: - [[Title]] — link to entry "Title", display "Title" - [[Title|display]] — link to entry "Title", display "display" - [[Title#anchor]] — link to a section anchor in "Title" - [[Title#anchor|display]] — link to anchor with custom display text Parsing rules: split on the FIRST "|" to separate target from display (the display may itself contain "|"); split the target on the FIRST "#" to separate page-title from anchor. The anchor is normalized via Hugo's anchorize to match heading slugs. Title and aliases are matched case-insensitively (resolve-link.html). Code regions (
, inline ) AND math regions ($...$, $$...$$,
  \(...\), \[...\]) are stashed behind placeholders before the rewrite
  and restored afterwards. This protects literal "[[ ]]" inside code
  blocks (so the wikilink syntax can be shown as-is) and inside math
  (e.g. the formal-power-series notation k[[x]]).
*/ -}}

...
 blocks (multiline). */ -}}
{{- $preBlocks := findRE `(?s)]*>.*?
` $content -}} {{- range $i, $block := $preBlocks -}} {{- $token := printf "@@WIKILINK_PRE_%d@@" $i -}} {{- $content = replace $content $block $token 1 -}} {{- end -}} {{- /* (2) Stash inline ... blocks. */ -}} {{- $codeBlocks := findRE `]*>[^<]*` $content -}} {{- range $i, $block := $codeBlocks -}} {{- $token := printf "@@WIKILINK_CODE_%d@@" $i -}} {{- $content = replace $content $block $token 1 -}} {{- end -}} {{- /* (3a) Stash display-math \[...\] (multiline). */ -}} {{- $mathDisplaySq := findRE `(?s)\\\[.+?\\\]` $content -}} {{- range $i, $block := $mathDisplaySq -}} {{- $token := printf "@@WIKILINK_MDSQ_%d@@" $i -}} {{- $content = replace $content $block $token 1 -}} {{- end -}} {{- /* (3b) Stash display-math $$...$$ (multiline). */ -}} {{- $mathDisplayDollar := findRE `(?s)\$\$.+?\$\$` $content -}} {{- range $i, $block := $mathDisplayDollar -}} {{- $token := printf "@@WIKILINK_MDD_%d@@" $i -}} {{- $content = replace $content $block $token 1 -}} {{- end -}} {{- /* (3c) Stash inline-math \(...\) (single line). */ -}} {{- $mathInlinePar := findRE `\\\(.+?\\\)` $content -}} {{- range $i, $block := $mathInlinePar -}} {{- $token := printf "@@WIKILINK_MIP_%d@@" $i {{- range $i, $block := $mathInlineDollar -}} {{- $token := printf "@@WIKILINK_MID_%d@@" $i -}} {{- $content = replace $content $block $token 1 -}} {{- end -}} {{- /* (4) Resolve each [[...]] form. */ -}} {{- $matches := findRE `\[\[[^\[\]]+\]\]` $content -}} {{- range uniq $matches -}} {{- $whole := . -}} {{- $inner := replaceRE `(?s)^\[\[(.+)\]\]$` `$1` $whole -}} {{- $pipeParts := split $inner "|" -}} {{- $target := index $pipeParts 0 -}} {{- $hasDisplay := gt (len $pipeParts) 1 -}} {{- $display := "" -}} {{- if $hasDisplay -}} {{- $display = delimit (after 1 $pipeParts) "|" -}} {{- else -}} {{- $display = trim $target " " -}} {{- end -}} {{- $hashParts := split $target "#" -}} {{- $pageTitle := replaceRE `\s+` " " (trim (index $hashParts 0) " ") -}} {{- $anchor := "" -}} {{- if gt := $mathInlinePar -}} {{- $token := printf "@@WIKILINK_MIP_%d@@" $i -}} {{- $content = replace $content $token $block -}} {{- end -}} {{- range $i, $block := $mathDisplayDollar -}} {{- $token := printf "@@WIKILINK_MDD_%d@@" $i -}} {{- $content = replace $content $token $block -}} {{- end -}} {{- range $i, $block := $mathDisplaySq -}} {{- $token := printf "@@WIKILINK_MDSQ_%d@@" $i -}} {{- $content = replace $content $token $block -}} {{- end -}} {{- range $i, $block := $codeBlocks -}} {{- $token := printf "@@WIKILINK_CODE_%d@@" $i -}} {{- $content = replace $content $token $block -}} {{- end -}} {{- range $i, $block := $preBlocks -}} {{- $token := printf "@@WIKILINK_PRE_%d@@" $i -}} {{- $content = replace $content $token $block -}} {{- end -}} {{- return $content -}}
