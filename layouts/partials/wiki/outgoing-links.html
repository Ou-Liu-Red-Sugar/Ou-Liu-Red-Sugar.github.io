{{- /* layouts/partials/wiki/outgoing-links.html Input : a Page (the current wiki entry). Output : an HTML block listing the wiki entries this page references via [[Title]] in its raw markdown. Renders nothing if the entry contains no resolvable wikilinks. */ -}} {{- $current := . -}} {{- $entries := slice -}} {{- range where site.RegularPages "Section" "wiki" -}} {{- if and (ne .RelPermalink $current.RelPermalink) (ne .RelPermalink "/wiki/graph/") -}} {{- $entries = $entries | append . -}} {{- end -}} {{- end -}} {{- /* Title-and-alias lookup over the OTHER wiki entries */ -}} {{- $byTitle := dict -}} {{- range $entries -}} {{- $page := . -}} {{- $byTitle = merge $byTitle (dict (lower $page.Title) $page) -}} {{- range $page.Params.aliases -}} {{- $byTitle = merge $byTitle (dict (lower .) $page) -}} {{- end -}} {{- end -}} {{- /* Resolve wikilink references in this page's RawContent. Strip the outer brackets, drop any |display tail (everything after the first |), drop any #anchor tail (everything after the first #), and look up the remaining title. */ -}} {{- $matches := findRE `\[\[[^\[\]]+\]\]` $current.RawContent -}} {{- $resolved := slice -}} {{- range uniq $matches -}} {{- $inner := replaceRE `(?s)^\[\[(.+)\]\]$` `$1` . -}} {{- $target := index (split $inner "|") 0 -}} {{- $title := lower (replaceRE `\s+` " " (trim (index (split $target "#") 0) " ")) -}} {{- with index $byTitle $title -}} {{- $resolved = $resolved | append . -}} {{- end -}} {{- end -}} {{- $resolved = sort (uniq $resolved) "Title" "asc" -}} {{- if gt (len $resolved) 0 -}}

{{- end -}}
