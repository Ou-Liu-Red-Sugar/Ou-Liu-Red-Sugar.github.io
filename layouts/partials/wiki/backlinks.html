{{- /* layouts/partials/wiki/backlinks.html Input : a Page (the current wiki entry). Output : an HTML block listing wiki entries that link to it via [[Title]] (matching this page's title or any of its aliases). Renders nothing if there are no backlinks. Designed to sit inside the Neighborhood section. */ -}} {{- $current := . -}} {{- $targets := slice (lower $current.Title) -}} {{- range $current.Params.aliases -}} {{- $targets = $targets | append (lower .) -}} {{- end -}} {{- $backlinks := slice -}} {{- range where site.RegularPages "Section" "wiki" -}} {{- $other := . -}} {{- if and (ne $other.RelPermalink $current.RelPermalink) (ne $other.RelPermalink "/wiki/graph/") -}} {{- $matches := findRE `\[\[[^\[\]]+\]\]` $other.RawContent -}} {{- $hit := false -}} {{- range uniq $matches -}} {{- if not $hit -}} {{- $title := lower (trim (replaceRE `^\[\[(.+)\]\]$` `$1` .) " ") -}} {{- if in $targets $title -}} {{- $hit = true -}} {{- end -}} {{- end -}} {{- end -}} {{- if $hit -}} {{- $backlinks = $backlinks | append $other -}} {{- end -}} {{- end -}} {{- end -}} {{- $backlinks = sort (uniq $backlinks) "Title" "asc" -}} {{- if gt (len $backlinks) 0 -}}

  • {{ .Title }} {{- with .Params.description }} — {{ . | markdownify }} {{- end }}

{{- end -}}
