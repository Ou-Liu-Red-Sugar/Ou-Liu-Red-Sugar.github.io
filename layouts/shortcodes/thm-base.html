{{- /* Shared renderer for theorem-like environments with numbering and references */ -}} {{- define "thm-block" -}} {{- $ctx := .ctx -}} {{- $kind := .kind -}} {{- $page := $ctx.Page -}} {{- $pageKey := $page.RelPermalink -}} {{- $counterKey := printf "thm_counter:%s:%s" $pageKey (lower $kind) -}} {{- $registryKey := printf "thm_registry:%s" $pageKey -}} {{- $title := $ctx.Get "title" -}} {{- $idParam := $ctx.Get "id" -}} {{- $star := false -}} {{- range $key := slice "star" "starred" "unnumbered" -}} {{- with $ctx.Get $key -}} {{- if or (eq . true) (eq (printf "%v" .) "true") -}} {{- $star = true -}} {{- end -}} {{- end -}} {{- end -}} {{- $number := "" -}} {{- if not $star -}} {{- $current := default 0 ($page.Store.Get $counterKey) -}} {{- $next := add $current 1 -}} {{- $page.Store.Set $counterKey $next -}} {{- $number = $next -}} {{- end -}} {{- $anchorBase := "" -}} {{- if $idParam -}} {{- $anchorBase = $idParam -}} {{- else if and (not $star) $number -}} {{- $anchorBase = printf "%s-%v" (lower $kind) $number -}} {{- end -}} {{- $anchor := "" -}} {{- if $anchorBase -}} {{- $anchor = anchorize $anchorBase -}} {{- end -}} {{- $registry := default (dict) ($page.Store.Get $registryKey) -}} {{- if $idParam -}} {{- $entry := dict "kind" $kind "number" $number "star" $star "anchor" $anchor -}} {{- $registry = merge $registry (dict $idParam $entry) -}} {{- end -}} {{- $page.Store.Set $registryKey $registry -}}
{{ $kind }} {{- if and (not $star) $number }} {{ $number }}{{ end -}} {{- with $title }} ({{ . }}){{ end -}}.
{{- $ctx.Inner | $ctx.Page.RenderString -}}
{{- end -}}
